12def dist(cor): 

13 """ 

14 Compute the correlation based distance matrix d, 

15 compare with page 239 of the first book by Marcos 

16 :param cor: the n x n correlation matrix 

17 :return: The matrix d indicating the distance between column i and i. 

18 Note that all the diagonal entries are zero. 

19 

20 """ 

21 # https://stackoverflow.com/questions/18952587/ 

22 matrix = np.sqrt(np.clip((1.0 - cor) / 2.0, a_min=0.0, a_max=1.0)) 

23 np.fill_diagonal(matrix, val=0.0) 

24 return ssd.squareform(matrix)

27def linkage(dist_vec, method="ward", **kwargs): 

28 """ 

29 Based on distance matrix compute the underlying links 

30 :param dist_vec: The distance vector based on the correlation matrix 

31 :param method: "single", "ward", etc. 

32 :return: links The links describing the graph (useful to draw the dendrogram) 

33 and basis for constructing the tree object 

34 """ 

35 # compute the root node of the dendrogram 

36 return sch.linkage(dist_vec, method=method, **kwargs)

39def tree(links): 

40 """ 

41 Compute the root ClusterNode. 

42 :param links: The Linkage matrix compiled by the linkage function above 

43 :return: The root node. From there it's possible to reach the entire graph 

44 """ 

45 return sch.to_tree(links, rd=False)

48def build_cluster(node, cov): 

49 """compute a cluster""" 

50 if node.is_leaf(): 

51 # a node is a leaf if has no further relatives downstream. 

52 # no leaves, no branches, ... 

53 asset = cov.keys().to_list()[node.id] 

54 return Cluster(assets={asset: 1.0}, variance=cov[asset][asset]) 

55 

56 # drill down on the left 

57 cluster_left = build_cluster(node.left, cov) 

58 # drill down on the right 

59 cluster_right = build_cluster(node.right, cov) 

60 # combine left and right into a new cluster 

61 return risk_parity(cluster_left, cluster_right, cov=cov)

64def hrp(prices, node=None, method="single"): 

65 """ 

66 Computes the root node for the hierarchical risk parity portfolio 

67 :param cov: This is the covariance matrix that shall be used 

68 :param node: Optional. This is the rootnode of the graph describing the dendrogram 

69 :return: the root cluster of the risk parity portfolio 

70 """ 

71 returns = prices.pct_change().dropna(axis=0, how="all") 

72 cov, cor = returns.cov(), returns.corr() 

73 node = node or tree(linkage(dist(cor.values), method=method)) 

74 

75 return build_cluster(node, cov)
